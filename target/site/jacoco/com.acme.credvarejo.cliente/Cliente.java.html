	public Cliente(Cpf cpf, String nome, int idade, Date clienteDesde, double renda, int sexo) {
		this.cpf = cpf;
		this.nome = nome;

	public String getPrimeiroNome() {
		String[] textoSeparado = nome.split(" ");

		return textoSeparado[0];
	}

		String[] textoSeparado = nome.split(" ");
